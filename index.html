// 在它加载完成后，执行下面的方法 function load() { console.log('loaded') activateTheme("light"); } // 这里我们封装了一个activateTheme方法，方便后边复用，作用是修改iframe内部的主题颜色 function activateTheme(theme) { var iframe = document.getElementById("i3"); if (iframe && iframe.contentWindow) { iframe.contentWindow.postMessage(theme, "*"); } }
